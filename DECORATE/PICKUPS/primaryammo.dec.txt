Actor SmallPrimaryAmmo : CustomInventory replaces clip
{
	Inventory.PickupMessage "Picked up some primary ammo."
	Inventory.PickupSound "Pickups/AMPIK"
	+COUNTITEM
	states
	{
	Spawn:
		PABX A -1
		Stop
	Pickup:
		TNT1 A 0 A_GiveInventory("UACAmmo", 10)
		TNT1 A 0 A_GiveInventory("Shells", 4)
		TNT1 A 0 A_GiveInventory("RhinoAmmo", 2)
		TNT1 A 0 A_GiveInventory("SMGAmmo", 10)
		stop
	}
}
