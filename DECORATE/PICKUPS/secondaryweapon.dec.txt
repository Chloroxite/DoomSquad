Actor SecondaryPickupReplaceRL : CustomInventory Replaces RocketLauncher
{
	Inventory.PickupMessage ""
	Inventory.PickupSound ""
	+COUNTITEM
	states
	{
	Spawn:
		SCBX A 0
		SCBX A 0 A_JumpIf(ACS_NamedExecuteWithResult("WasIDropped") == 1, "IWasDroppedOnMyHead")
		SCBX A 35 A_JumpIf(1 == 1, "SpriteLoop")
		Wait
	SpriteLoop:
		SCBX A 1 A_ChangeFlag("FLOATBOB", 0)
		Loop
	IWasDroppedOnMyHead:
		TNT1 A 0 A_Print("This item has been dropped")
		SCBX A 1 A_SpawnItemEX("SecondaryPickupDrop", 0, 0, 0, momx, momy, momz, 0, SXF_ABSOLUTEMOMENTUM)
		Stop
	Pickup:
		TNT1 A 0 A_JumpIf((ACS_NamedExecuteWithResult("DS_GiveWeapons", 3, 0) & 3) == 1, "DespawnPickup")
		Goto RetainPickup
	RetainPickup:
		TNT1 A 0 A_ChangeFlag("FLOATBOB", 0) //Hehe funny story: For some reason changing this random ass flag makes this work...
		Fail
	DespawnPickup:
		TNT1 A 0 A_ChangeFlag("FLOATBOB", 0)
		Stop
	}
}

Actor SecondaryPickupDrop : SecondaryPickupReplaceRL
{
	states
	{
	Spawn:
		TNT1 A 0
		Goto SpriteLoop
	Pickup:
		TNT1 A 0 A_JumpIf((ACS_NamedExecuteWithResult("DS_GiveWeapons", 3, 1) & 1) == 1, "DespawnPickup")
		Goto RetainPickup
	}
}
