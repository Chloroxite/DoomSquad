#library "DSFuncs"
#include "zcommon.acs"

script "WasIDropped" (void)
{
	setResultValue(GetActorProperty(0, APROP_DROPPED));
}

function int GetClassNumber (void)
{
	int classnum = 0;
	if(CheckInventory("IsAssault"))
		classnum = 0;
	if(CheckInventory("IsMedic"))
		classnum = 1;
	if(CheckInventory("IsSharpshooter"))
		classnum = 2;
	if(CheckInventory("IsSupport"))
		classnum = 3;
	
	return classnum;
}
